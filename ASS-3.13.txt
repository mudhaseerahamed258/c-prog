#include <stdio.h>

int main() {
    int rows = 5; 
    
    // loop for each row
    for (int i = 1; i <= rows; i++) {
        // loop for spaces
        for (int j = 1; j <= rows - i; j++) {
            printf(" ");
        }
      
        for (int k = 1; k <= i; k++) {
            if (k == i || k == 1 || i == rows) {
                printf("* ");
            } else {
                printf("  ");
            }
        }
       
        for (int l = i + 1; l <= rows; l++) {
            if (l == i + 1 || l == rows || i == rows) {
                printf("* ");
            } else {
                printf("  ");
            }
        }
        printf("\n");
    }
    
    return 0;
}
