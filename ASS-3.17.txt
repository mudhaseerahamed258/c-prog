#include <stdio.h>

int main() {
    int arr[] = {5, -3, 0, -8, 2, -1};
    int n = sizeof(arr)/sizeof(arr[0]);
    int count = 0;

    for(int i=0; i<n; i++) {
        if(arr[i] < 0) {
            count++;
        }
    }

    printf("Number of negative numbers in the array: %d\n", count);

    return 0;
}
